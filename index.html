// SPDX-License-Identifier: MIT pragma solidity ^0.8.24; /// @title VaultDefender — an on-chain arcade shooter for the Arc network /// @notice Players stake the native token (USDC on Arc) to enter a ranked /// round, defend the Vault against waves of enemies in the /// frontend, then submit their score. The current #1 defender can /// claim the accumulated prize pool and receives a Trophy NFT. contract VaultDefender { // --------------------------------------------------------------- // Config // --------------------------------------------------------------- /// @notice Cost to enter one ranked round, in the chain's native token /// (USDC has 6 decimals on Arc, so 100_000 == 0.10 USDC) uint256 public constant ENTRY_FEE = 100_000; uint256 public constant LEADERBOARD_SIZE = 10; // --------------------------------------------------------------- // Game state // --------------------------------------------------------------- struct Entry { address player; uint256 score; uint256 timestamp; } mapping(address => uint256) public bestScore; mapping(address => uint256) public roundsPlayed; mapping(address => bool) public hasActiveRound; mapping(address => uint256) public trophyCount; Entry[LEADERBOARD_SIZE] public leaderboard; uint256 public leaderboardLength; uint256 public prizePool; bool private _locked; // --------------------------------------------------------------- // Trophy NFT (minimal ERC721, no external imports) // --------------------------------------------------------------- string public constant name = "Vault Defender Trophy"; string public constant symbol = "VDT"; uint256 public nextTokenId = 1; mapping(uint256 => address) private _owners; mapping(address => uint256) private _balances; mapping(uint256 => address) private _tokenApprovals; mapping(address => mapping(address => bool)) private _operatorApprovals; mapping(uint256 => uint256) public trophyScore; // --------------------------------------------------------------- // Events // --------------------------------------------------------------- event RoundEntered(address indexed player, uint256 fee, uint256 newPoolTotal); event ScoreSubmitted(address indexed player, uint256 score, uint256 timestamp); event NewPersonalBest(address indexed player, uint256 oldBest, uint256 newBest); event LeaderboardUpdated(address indexed player, uint256 score, uint256 rank); event PrizeClaimed(address indexed player, uint256 amount, uint256 tokenId); event Transfer(address indexed from, address indexed to, uint256 indexed tokenId); event Approval(address indexed owner, address indexed approved, uint256 indexed tokenId); event ApprovalForAll(address indexed owner, address indexed operator, bool approved); modifier nonReentrant() { require(!_locked, "reentrant call"); _locked = true; _; _locked = false; } // --------------------------------------------------------------- // Game logic // --------------------------------------------------------------- /// @notice Pay the entry fee to start a ranked round. Must be followed /// by exactly one submitScore() call once the round ends. function enterRound() external payable { require(msg.value == ENTRY_FEE, "send exact ENTRY_FEE"); require(!hasActiveRound[msg.sender], "round already active"); hasActiveRound[msg.sender] = true; roundsPlayed[msg.sender] += 1; prizePool += msg.value; emit RoundEntered(msg.sender, msg.value, prizePool); } /// @notice Submit the score for the currently active ranked round. function submitScore(uint256 score) external { require(hasActiveRound[msg.sender], "no active round - call enterRound first"); require(score > 0, "score must be > 0"); hasActiveRound[msg.sender] = false; emit ScoreSubmitted(msg.sender, score, block.timestamp); uint256 currentBest = bestScore[msg.sender]; if (score > currentBest) { bestScore[msg.sender] = score; emit NewPersonalBest(msg.sender, currentBest, score); } _tryInsertLeaderboard(msg.sender, score); } /// @notice Only the current #1 defender can claim the whole prize pool. /// Claiming resets the pool to zero and mints a Trophy NFT. function claimPrize() external nonReentrant { require(leaderboardLength > 0, "leaderboard is empty"); require(leaderboard[0].player == msg.sender, "only the top defender can claim"); require(prizePool > 0, "prize pool is empty"); uint256 amount = prizePool; prizePool = 0; uint256 tokenId = _mintTrophy(msg.sender, leaderboard[0].score); (bool ok, ) = msg.sender.call{value: amount}(""); require(ok, "prize transfer failed"); emit PrizeClaimed(msg.sender, amount, tokenId); } function getLeaderboard() external view returns (Entry[LEADERBOARD_SIZE] memory) { return leaderboard; } function getPlayerStats(address player) external view returns (uint256 best, uint256 rounds, uint256 trophies, bool activeRound) { return (bestScore[player], roundsPlayed[player], trophyCount[player], hasActiveRound[player]); } function _tryInsertLeaderboard(address player, uint256 score) internal { if (leaderboardLength == LEADERBOARD_SIZE && score <= leaderboard[LEADERBOARD_SIZE - 1].score) { return; } uint256 len = leaderboardLength; for (uint256 i = 0; i < len; i++) { if (leaderboard[i].player == player) { for (uint256 j = i; j < len - 1; j++) { leaderboard[j] = leaderboard[j + 1]; } len -= 1; delete leaderboard[len]; break; } } uint256 insertAt = len; for (uint256 i = 0; i < len; i++) { if (score > leaderboard[i].score) { insertAt = i; break; } } if (insertAt >= LEADERBOARD_SIZE) { leaderboardLength = len; return; } uint256 newLen = len < LEADERBOARD_SIZE ? len + 1 : LEADERBOARD_SIZE; for (uint256 k = newLen - 1; k > insertAt; k--) { leaderboard[k] = leaderboard[k - 1]; } leaderboard[insertAt] = Entry({player: player, score: score, timestamp: block.timestamp}); leaderboardLength = newLen; emit LeaderboardUpdated(player, score, insertAt); } // --------------------------------------------------------------- // Minimal ERC721 implementation (trophy badges only, non-transferable // logic kept simple: standard transfer/approval flow still works so // trophies remain compatible with wallets and explorers) // --------------------------------------------------------------- function _mintTrophy(address to, uint256 scoreAtMint) internal returns (uint256 tokenId) { tokenId = nextTokenId++; _owners[tokenId] = to; _balances[to] += 1; trophyScore[tokenId] = scoreAtMint; trophyCount[to] += 1; emit Transfer(address(0), to, tokenId); } function balanceOf(address owner) external view returns (uint256) { require(owner != address(0), "zero address"); return _balances[owner]; } function ownerOf(uint256 tokenId) public view returns (address owner) { owner = _owners[tokenId]; require(owner != address(0), "token does not exist"); } function approve(address to, uint256 tokenId) external { address owner = ownerOf(tokenId); require(to != owner, "approve to owner"); require(msg.sender == owner || _operatorApprovals[owner][msg.sender], "not authorized"); _tokenApprovals[tokenId] = to; emit Approval(owner, to, tokenId); } function getApproved(uint256 tokenId) external view returns (address) { ownerOf(tokenId); return _tokenApprovals[tokenId]; } function setApprovalForAll(address operator, bool approved) external { require(operator != msg.sender, "approve to self"); _operatorApprovals[msg.sender][operator] = approved; emit ApprovalForAll(msg.sender, operator, approved); } function isApprovedForAll(address owner, address operator) external view returns (bool) { return _operatorApprovals[owner][operator]; } function transferFrom(address from, address to, uint256 tokenId) public { require(ownerOf(tokenId) == from, "from is not owner"); require(to != address(0), "transfer to zero address"); require( msg.sender == from || _tokenApprovals[tokenId] == msg.sender || _operatorApprovals[from][msg.sender], "not authorized" ); delete _tokenApprovals[tokenId]; _balances[from] -= 1; _balances[to] += 1; _owners[tokenId] = to; emit Transfer(from, to, tokenId); } function safeTransferFrom(address from, address to, uint256 tokenId) external { safeTransferFrom(from, to, tokenId, ""); } function safeTransferFrom(address from, address to, uint256 tokenId, bytes memory data) public { transferFrom(from, to, tokenId); if (to.code.length > 0) { (bool ok, bytes memory ret) = to.call( abi.encodeWithSelector( bytes4(keccak256("onERC721Received(address,address,uint256,bytes)")), msg.sender, from, tokenId, data ) ); require(ok, "receiver call failed"); bytes4 retval = abi.decode(ret, (bytes4)); require(retval == bytes4(keccak256("onERC721Received(address,address,uint256,bytes)")), "unsafe receiver"); } } function tokenURI(uint256 tokenId) external view returns (string memory) { ownerOf(tokenId); return string( abi.encodePacked( "data:application/json,%7B%22name%22:%22Vault%20Defender%20Trophy%20%23", _toString(tokenId), "%22,%22description%22:%22Awarded%20for%20topping%20the%20Vault%20Defender%20leaderboard%20on%20Arc%22,%22attributes%22:%5B%7B%22trait_type%22:%22Score%22,%22value%22:", _toString(trophyScore[tokenId]), "%7D%5D%7D" ) ); } function supportsInterface(bytes4 interfaceId) external pure returns (bool) { return interfaceId == 0x80ac58cd || interfaceId == 0x01ffc9a7; // ERC721, ERC165 } function _toString(uint256 value) internal pure returns (string memory) { if (value == 0) return "0"; uint256 temp = value; uint256 digits; while (temp != 0) { digits++; temp /= 10; } bytes memory buffer = new bytes(digits); while (value != 0) { digits -= 1; buffer[digits] = bytes1(uint8(48 + (value % 10))); value /= 10; } return string(buffer); } }
